# eq (equal)

> eq(column: string, param: Param)

```ts
import { OP, Param } from 'mysql2-orm';

/**
 * { condition: '`id` = ?', params: [16] }
 */
OP.eq('id', 16);

/**
 * { condition: '`name` = ?', params: ['John'] }
 */
OP.eq('name', 'John');
```
